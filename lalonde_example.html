data("lalonde")

## outcome of interest and treatment indicator

Y <- lalonde$re78
Tr <- lalonde$treat
  
# estimate first propensity model using a glm

glm1 <- glm(
  Tr ~ 
    age +
    educ +
    black + 
    hisp +
    married +
    nodegr + 
    re74 +
    re75,
  family = binomial,
  data = lalonde
)

## Now one to one matching with replacement

rr1 <- Match(
  Y = Y, 
  Tr = Tr, 
  X = glm1$fitted,
  ties = TRUE,
  estimand = "ATT"
)

## See results from Match
